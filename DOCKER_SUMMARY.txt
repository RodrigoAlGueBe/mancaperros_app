================================================================================
                          VERIFICACIÓN DE REQUISITOS
================================================================================

✓ Python 3.12              -> FROM python:3.12-slim
✓ Multi-stage build        -> Builder + Runtime stages
✓ Usuario no-root          -> RUN groupadd -r appuser && useradd -r -g appuser appuser
✓ requirements.txt         -> COPY requirements.txt . (Stage 1)
✓ Instalar dependencias    -> pip install --no-cache-dir -r requirements.txt
✓ Puerto 8000              -> EXPOSE 8000
✓ Comando uvicorn          -> CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]

================================================================================
                          CÓMO USAR LOS SCRIPTS
================================================================================

LINUX/MAC:
  chmod +x build.sh
  ./build.sh                      # Build con tag 'latest'
  ./build.sh v1.0.0               # Build con tag v1.0.0
  ./build.sh v1.0.0 ghcr.io/myorg # Build y push a registry

WINDOWS POWERSHELL:
  .\build.ps1                          # Build con tag 'latest'
  .\build.ps1 -Tag v1.0.0              # Build con tag v1.0.0
  .\build.ps1 -Tag v1.0.0 -Registry ghcr.io/myorg  # Build y push

MANUAL (CUALQUIER PLATAFORMA):
  docker build -t mancaperros_app:latest .

================================================================================
                          EJECUTAR CONTENEDOR
================================================================================

BÁSICO:
  docker run -d -p 8000:8000 mancaperros_app:latest

CON VARIABLES DE ENTORNO:
  docker run -d \
    -p 8000:8000 \
    -e DATABASE_URL=mysql://user:pass@host:3306/db \
    mancaperros_app:latest

CON ARCHIVO .env:
  docker run -d -p 8000:8000 --env-file .env mancaperros_app:latest

PRODUCCIÓN:
  docker run -d \
    --name mancaperros \
    --restart unless-stopped \
    -p 8000:8000 \
    --env-file .env \
    -v mancaperros_data:/app/data \
    mancaperros_app:latest
